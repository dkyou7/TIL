# 알고리즘 공부한거

1. [에라토스테네스의 체](https://marobiana.tistory.com/91)

   - ```c++
     #include<iostream>
     using namespace std;
     
     void getche(int num) {
     	int* arr = new int[num];
     	for (int i = 2; i <= num; i++) {
     		arr[i] = i;
     	}
     
     	// 배수는 모두 0으로 처리해준다. 체를 거르는 것처럼 걸른다.
     	for (int i = 2; i <= num; i++) {
     		if (arr[i] == 0) {
     			continue;
     		}
     		// 배수는 다 걸른다. 0으로 만듬.
     		for (int j = i*2; j <= num; j+=i) {
     			arr[j] = 0;
     		}
     	}
     	for (int i = 2; i <= num; i++) {
     		if(arr[i]!=0)
     			cout << arr[i] << " ";
     	}
     }
     int main() {
     	int num;
     	cin >> num;
     	getche(num);
     
     	return 0;
     }
     ```
